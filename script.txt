pipeline {
    agent any
    tools{
        maven 'MAVEN-HOME'
    }
    stages {
        stage('git repo & clean') {
            steps {
                bat "git clone https://github.com/nikhileshnilagiri/maven-web-app"
                bat "mvn clean -f maven-web-app/pom.xml"
            }
        }
        stage('install') {
            steps {
                bat "mvn install -f maven-web-app/pom.xml"
            }
        }
        stage('test') {
            steps {
                bat "mvn test -f maven-web-app/pom.xml"
            }
        }
        stage('package') {
            steps {
                bat 'mvn package -f maven-web-app/pom.xml'
            }
        }
    }
}
